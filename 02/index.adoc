= HW2: Text Mining

== Run and deactivate

[source, bash]
----
# Create env
virtualenv ddw-hw-2

# Activate env
source ddw-hw-2/bin/activate

# Install requirements and run the jupyter server
./bin/run.sh

# Deactivate env
deactivate

# Remove env
./bin/remove.sh

----
